    # File tests/tile_rack/test_has_tiles_for.rb, line 63
def test_rack_contains_a_word_with_duplicate_letters
  @rack.append(:S)
  @rack.append(:E)
  @rack.append(:X)
  @rack.append(:C)
  @rack.append(:E)
  @rack.append(:S)
  @rack.append(:S)
  assert_equal true, @rack.has_tiles_for?("excess")
end
