def test_that_full_tile_rack_doesnt_need_any_tiles
  @rack.append(:I)
  @rack.append(:T)
  @rack.append(:S)
  @rack.append(:F)
  @rack.append(:U)
  @rack.append(:L)
  @rack.append(:L)
  assert_equal 7, @rack.tiles.length
  assert_equal 0, @rack.number_of_tiles_needed
end
